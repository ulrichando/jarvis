# LiveKit Agents + Groq voice stack. Pinned to minor versions that
# were live when the agent was first wired; bump these deliberately.
#
# Install:
#   python3 -m venv .venv
#   source .venv/bin/activate
#   pip install -r requirements.txt

livekit-agents[silero,openai,groq]~=1.5

# Convex client used for the conversations.db mirror. Mirror is best-
# effort: if the package is missing or the backend is unreachable, the
# agent logs a warning and continues SQLite-only. See _get_convex_client
# in jarvis_agent.py.
convex~=0.7

# Gemini vision for desktop computer-use (screen description). The
# voice agent calls Gemini Flash with a screenshot to get a structured
# UI description, then Groq/DeepSeek decides the next click/type.
google-genai~=1.73

# Face ID — InsightFace buffalo_l (RetinaFace + ArcFace ResNet-50),
# the production-grade open-source face stack used by AWS Rekognition
# clones, ID-verification SaaS, etc. 512-d L2-normalized embeddings,
# cosine distance, ~99.83% LFW. ONNX Runtime executes the models on CPU.
# Models (~280 MB) auto-download to ~/.insightface/ on first use.
# Multi-frame averaged enrollment + IR/RGB cross-camera anti-spoof are
# implemented in jarvis_computer_use.py:face_register/face_identify.
insightface~=0.7
onnxruntime~=1.20
# pkg_resources (transitively required by older deps); pin to a setuptools
# version that still ships it (82+ removed pkg_resources).
setuptools<70
