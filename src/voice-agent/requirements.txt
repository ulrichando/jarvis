# LiveKit Agents + Groq voice stack. Pinned to minor versions that
# were live when the agent was first wired; bump these deliberately.
#
# Install:
#   python3 -m venv .venv
#   source .venv/bin/activate
#   pip install -r requirements.txt

livekit-agents[silero,openai,groq,anthropic]~=1.5

# Deepgram Nova-3 streaming STT — primary STT path as of 2026-05-18
# for fast barge-in. Without this, Groq Whisper Turbo (final-only)
# made STT-confirmed barge-in unusable (no partials = no real-time
# turn detection). Requires DEEPGRAM_API_KEY in .env. Falls through
# to Whisper-only when the key is unset (`providers/stt.py` handles
# the absence gracefully).
livekit-plugins-deepgram~=1.5

# Agent Client Protocol — JSON-RPC over stdio so IDEs (Zed, Cursor,
# VS Code via plugin) can drive JARVIS as a coding agent. Loaded by
# the `acp_adapter/` package and `bin/jarvis-acp`. Pinned to 0.9.0
# because newer versions reshape the schema in breaking ways the
# adapter hasn't been re-validated against.
agent-client-protocol==0.9.0

# Microsoft Edge-TTS — neural voices reachable without auth/quota.
# Wrapped in providers/edge_tts.py as a livekit-agents TTS adapter.
# Imported at the top of jarvis_agent.py, so a missing dep crashes the
# entire agent at import time (live failure 2026-05-15 — first run
# after fresh install died here because the import was added before
# anyone added the package to requirements).
edge-tts~=7.0

# Playwright — drives a bundled Chromium via CDP for the browser
# subagent's fallback path. When the Chrome extension isn't connected
# (cold Chrome, extension crashed, user disabled it), tools/browser_cdp.py
# takes over and the subagent stays usable. Pinned to major 1.x; the
# CDP surface and async API have been stable since 1.40+. Adding this
# also requires running `playwright install chromium` to fetch the
# ~200MB browser binary — install.sh prompts for confirmation since
# it's a one-time large download. Skip-flag: JARVIS_SKIP_CDP=1.
playwright>=1.49.0,<2.0

# systemd notify-protocol helper. jarvis-voice-agent.service is
# Type=notify with WatchdogSec=120s — the unit fails to start at all
# if the process never sends READY=1, and gets killed mid-flight if
# it doesn't ping the watchdog. Imported in jarvis_agent.py:5358 and
# resilience/watchdog.py:19 (live failure 2026-05-15 second start).
sdnotify~=0.3

# Redis client for the JARVIS event hub (jarvis-hub.service). Hub-less
# operation is gracefully degraded — voice-agent logs 'hub disabled' and
# continues — but with redis installed, conversation mirror + cross-
# process event bus actually work. Pairs with redis-server.service.
redis~=5.0

# Vision backend has THREE paths:
#   1. Ollama (local, qwen2.5vl:3b) — no extra package.
#   2. Gemini (Google Generative Language) — primary cloud path.
#      Re-introduced 2026-05-11 evening after the morning's Kimi swap
#      proved too slow (~11s vs ~3-4s on gemini-2.5-flash-lite).
#   3. Kimi (Moonshot) — fallback when Gemini hits the API-disabled /
#      quota / 5xx markers. Uses openai-python with swapped base_url,
#      already in the dep graph below.
# 2.0+ required by livekit-plugins-google 1.5.6's RealtimeModel — it
# uses types.HistoryConfig in _build_connect_config when the model's
# mutable_chat_context=False (true for gemini-3.1-flash-live-preview).
# Without this bump, every Live session crashes with
# AttributeError: module 'google.genai.types' has no attribute 'HistoryConfig'
# (live failure 2026-05-11 13:14 UTC). browser-use 0.12.6 pins
# google-genai==1.65.0 — that pin is ignored; live testing confirms
# browser-use still works with 2.0.
google-genai>=2.0,<3.0

# Face ID — InsightFace buffalo_l (RetinaFace + ArcFace ResNet-50),
# the production-grade open-source face stack used by AWS Rekognition
# clones, ID-verification SaaS, etc. 512-d L2-normalized embeddings,
# cosine distance, ~99.83% LFW. ONNX Runtime executes the models on CPU.
# Models (~280 MB) auto-download to ~/.insightface/ on first use.
# Multi-frame averaged enrollment + IR/RGB cross-camera anti-spoof are
# implemented in jarvis_computer_use.py:face_register/face_identify.
insightface~=0.7
onnxruntime~=1.20
ai-edge-litert~=2.1  # DTLN-aec L3 neural echo cancellation (audio/dtln_aec.py)
# pkg_resources (transitively required by older deps); pin to a setuptools
# version that still ships it (82+ removed pkg_resources).
setuptools>=78.1.1,<82

# LangChain + LangGraph orchestration. The voice loop's per-turn dispatch
# is expressed as a StateGraph in turn_graph.py; the classifier uses
# LangChain's unified init_chat_model so the provider is pluggable
# (Groq default; DeepSeek/OpenAI/Anthropic via JARVIS_ROUTER_PROVIDER).
# Set JARVIS_GRAPH_DISABLED=1 to revert to the inline async dispatcher.
#
# Pinned with `~=` (semver-aware: accept patches + minors, refuse majors)
# so a 2.x release can't pip-install silently and break the graph.
langgraph~=1.0
langgraph-checkpoint-sqlite~=3.0
langchain~=1.0
langchain-groq~=1.0
langchain-openai~=1.2

# MCP (Model Context Protocol) client SDK. Powers tools/mcp_client.py, which
# connects to MCP servers configured in ~/.jarvis/mcp.json, discovers their
# tools, and registers each into the JARVIS tool registry (namespaced
# mcp__<server>__<tool>). The import in mcp_client.py is GUARDED — a missing
# package degrades the MCP tool surface to a clean no-op rather than crashing
# the agent, so this dep is functionally optional. Inert with no mcp.json.
mcp~=1.27

# Honcho cloud memory backend (plugins/memory/honcho/).
# The memory-provider layer is INERT without all three of:
#   1. this package installed
#   2. HONCHO_API_KEY set in .env
#   3. JARVIS_MEMORY_PROVIDER=honcho
# Missing any one → provider inert, file-memory unchanged, zero voice-turn impact.
# Pinned to the minor that was installed and tested (honcho-ai 2.1.1).
honcho-ai~=2.1

# psutil — cross-platform process inspection + termination. Used by
# tools/execute_code.py::_kill_process_group to walk the child process
# tree on POSIX (replaces os.killpg, which doesn't exist on Windows) and
# on Windows (where there's no SIGKILL either). Added 2026-05-24 as part
# of the Phase 2.1 cross-platform footgun cleanup.
psutil~=7.0

# pywinauto — Win32 GUI automation library used by tools/desktop_control.py's
# Windows backend (send_keys / find_windows / Application.connect for window
# activation + minimize / restore). Platform-conditional: installs ONLY on
# Windows hosts, so the Linux dev box's venv resolves cleanly without it.
# Lazy-imported inside each desktop_control helper so the module is import-
# safe even when the dep is absent. Added 2026-05-24 as part of the Phase
# 3.2 cross-platform desktop-control refactor.
pywinauto>=0.6.8; platform_system == "Windows"

cmudict~=1.1  # offline CMU pronunciation dictionary for viseme lip-sync (no model/torch)
vaderSentiment~=3.3  # offline lexicon sentiment for the face expression layer (no model/torch)
