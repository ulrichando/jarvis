# Test-only deps for the voice-agent pytest suite. Kept separate from
# requirements.txt so production installs don't pull in test fixtures.
#
# Install:
#   pip install -r requirements-test.txt
#
# CI runs this AFTER requirements.txt.

-r requirements.txt

# pytest-timeout — kill hung tests so CI never wedges.
pytest-timeout~=2.3

# pytest-asyncio — required for @pytest.mark.asyncio tests. Without this,
# async test bodies return a coroutine object (truthy) and pass vacuously.
# The TTS barge-in cancellation tests and dispatch_agent tests silently
# never ran their async code before this was added.
pytest-asyncio~=0.23
